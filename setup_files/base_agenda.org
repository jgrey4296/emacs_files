      (defun my-horizontal-recenter ()
        "make the point horizontally centered in the window"
        (interactive)
        (let ((mid (/ (window-width) 2))
              (line-len (save-excursion (end-of-line) (current-column)))
              (cur (current-column)))
          (if (< mid cur)
              (set-window-hscroll (selected-window)
                                  (- cur mid)))))
